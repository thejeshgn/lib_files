<script>
var term, ws;
ws = new WebSocket('ws://localhost:7890');
ws.onopen = function() {
lib.init(function() {
localStorage.clear()
term = new hterm.Terminal();
window.term = term;
term.decorate(document.getElementById('terminal'));
term.prefs_.set('audible-bell-sound', '')
term.prefs_.set('ctrl-c-copy', true);
term.prefs_.set('use-default-window-copy', true);
term.prefs_.set('font-size', 11)
term.prefs_.set('font-family', 'Consolas, Menlo, "Bitstream Vera Sans Mono", monospace')
term.prefs_.set('font-smoothing', 'none')
term.prefs_.set('cursor-color', 'rgba(100, 100, 100, 0.5)')
term.setCursorPosition(0, 0);
term.setCursorVisible(true);
ws.send(JSON.stringify({
args: document.location.hash.substr(1),
width: term.screenSize.width,
height: term.screenSize.height
}));
});
}
ws.onmessage = function(msg) {
msg && msg.data && term && term.io.writeUTF16(msg.data);
};
ws.onerror = function(e) {
console.log("WebSocket connection error");
};
ws.onclose = function() {
console.log("WebSocket connection closed");
};
</script>
